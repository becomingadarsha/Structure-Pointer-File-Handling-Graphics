//11.	Write a program which accept a series of integer numbers from the user 
//and stores all odd numbers in a file name odd.txt and all even numbers in a file name even.txt.
#include<stdio.h>
#include<string.h>
int main()
{
	FILE *fp1,*fp2;
	int i,n,a[100];
	printf("How many numbers?");
	scanf("%d",&n);
	printf("Enter numbers:");
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		
	}
	fp1=fopen("even.txt","w");
	fp2=fopen("odd.txt","w");
	for(i=1;i<=n;i++)
	{
		if((a[i]%2)==0)
		{
			fprintf(fp1,"%d\n",a[i]);
		}
		else
		{
			fprintf(fp2,"%d\n",a[i]);
		}
	}

	fclose(fp1);
	fclose(fp2);
}
